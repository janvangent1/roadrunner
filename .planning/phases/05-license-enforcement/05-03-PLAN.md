Update `getRoutesWithLicenseStatus()` to:
1. Call `licenseRepository.getMyLicenses()` first — this hydrates the cache. If it fails, proceed anyway (stale cache or empty — better than crashing)
2. Fetch routes via `apiService.getRoutes()`
3. For each route, call `licenseRepository.computeLicenseStatus(route.id)` (synchronous, reads from cache)
4. Build `RouteWithLicense` from those results

```kotlin
suspend fun getRoutesWithLicenseStatus(): Result<List<RouteWithLicense>> {
    return try {
        // Hydrate license cache first (failure is non-fatal — cache stays stale)
        licenseRepository.getMyLicenses()

        val routesResponse = apiService.getRoutes()
        if (!routesResponse.isSuccessful) {
            return Result.failure(Exception("Failed to load routes: ${routesResponse.code()}"))
        }
        val routes = routesResponse.body() ?: emptyList()
        val result = routes.map { route ->
            val (status, expiresAt, licenseType) = licenseRepository.computeLicenseStatus(route.id)
            RouteWithLicense(
                route = route,
                licenseStatus = status,
                expiresAt = expiresAt,
                licenseType = licenseType,
            )
        }
        Result.success(result)
    } catch (e: Exception) {
        Result.failure(e)
    }
}
```

fun loadRoute() {
    viewModelScope.launch {
        _uiState.update { it.copy(isLoading = true, error = null) }
        // Hydrate license cache first — ensures computeLicenseStatus has data
        licenseRepository.getMyLicenses()
        val routeResult = routeRepository.getRoute(routeId)

        routeResult
            .onSuccess { route ->
                _uiState.update {

                        licenseType = licenseResult.third,
                        isLoading = false,
                    )
                }
            }

}
